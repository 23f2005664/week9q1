Risk Calculation in Python


import numpy as np
import pandas as pd

# Value at Risk (95%)
def calculate_var(returns, confidence=0.95):
    return np.percentile(returns, (1 - confidence) * 100)

portfolio_returns = np.random.normal(0.001, 0.02, 1000)
var_95 = calculate_var(portfolio_returns)
print(f"95% VaR: {var_95:.4%}")
